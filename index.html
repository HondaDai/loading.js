Loading.js

A Simple Loading Overlay

Download this project as a .zip file Download this project as a tar.gz file

Usage

Show

$.loading.show();

$.loading.show({fadeIn: 1000, callback: function(){ ... } });

Hide

$.loading.hide();

$.loading.hide({fadeOut: 1000, callback: function(){ ... } });

Check Showing

$.loading.isShowing();

Demo

$.loading.show({callback: function(){ alert('show'); }});
setTimeout(function(){
  if ($.loading.isShowing())
    $.loading.hide({callback: function(){ alert('hide'); }});
}, 3000);
